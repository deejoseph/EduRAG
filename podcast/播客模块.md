# EduRAG 播客模块设计文档

## 一、模块概述

### 1.1 定位
基于 EduRAG 引导写作模块（审题→构思→写作→评估）四阶段输出，自动生成高质量高考作文辅导播客，实现 **"30分钟完成一篇高水平高考作文"** 的教学目标。

### 1.2 核心价值
| 维度 | 说明 |
|------|------|
| **效率** | 从题目到可发布播客，全流程自动化，一键生成 |
| **个性化** | 复用引导写作模块的148个题目+3万条RAG知识库，每期内容独特 |
| **真人人声** | 整合语音克隆技术，实现真人音色播客，非机械TTS |
| **多平台覆盖** | 一键分发至小宇宙、喜马拉雅、B站、Apple播客、Spotify |

### 1.3 用户画像
- **直接用户**：高三学生（每日通勤/洗漱/睡前15-20分钟碎片时间）
- **间接受众**：家长、语文教师（关注高考作文备考方法）

### 1.4 播客定位
- **名称建议**：《30分钟·高考作文实战》或《高考作文·极速通关》
- **目标**：每期20-25分钟，完成"审题→构思→范文带练→点评"全流程
- **更新频率**：日更（高考季5-7月可加密为日更2期）

# podcast/script_generator.py
from typing import Dict, Optional
import json
from datetime import datetime

class PodcastMaterialManager:
    """播客素材管理：从引导写作导入、存储、选择"""
    
    def __init__(self, storage_path: str = "./data/podcast_materials/"):
        self.storage_path = storage_path
        os.makedirs(storage_path, exist_ok=True)
    
    def import_from_writing(self, export_data: Dict) -> str:
        """从引导写作模块导入素材"""
        material_id = f"POD_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
        
        # 保存素材到本地
        file_path = os.path.join(self.storage_path, f"{material_id}.json")
        with open(file_path, 'w', encoding='utf-8') as f:
            json.dump({
                "id": material_id,
                "imported_at": datetime.now().isoformat(),
                "episode_title": export_data["episode_title"],
                "raw_materials": export_data["raw_materials"],
                "podcast_script": export_data["podcast_script"],
                "source_topic": export_data["source_topic"],
                "grade": export_data["grade"],
                "status": "imported"  # imported | editing | audio_generated | published
            }, f, ensure_ascii=False, indent=2)
        
        return material_id
    
    def list_materials(self, status: Optional[str] = None) -> List[Dict]:
        """列出所有已导入素材"""
        materials = []
        for file in os.listdir(self.storage_path):
            if not file.endswith('.json'):
                continue
            with open(os.path.join(self.storage_path, file), 'r') as f:
                data = json.load(f)
                if status is None or data.get("status") == status:
                    materials.append(data)
        return sorted(materials, key=lambda x: x["imported_at"], reverse=True)
    
    def get_material(self, material_id: str) -> Dict:
        """获取单个素材"""
        path = os.path.join(self.storage_path, f"{material_id}.json")
        with open(path, 'r') as f:
            return json.load(f)
    
    def update_material(self, material_id: str, updates: Dict):
        """更新素材（如润色后的文案）"""
        data = self.get_material(material_id)
        data.update(updates)
        data["updated_at"] = datetime.now().isoformat()
        with open(os.path.join(self.storage_path, f"{material_id}.json"), 'w') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

            if not config.get("enabled", True):
                continue
            if config["type"] == "api":
                clients[platform] = self._init_api_client(platform, config)
            elif config["type"] == "rss":
                clients[platform] = self._init_rss_client(platform, config)
            elif config["type"] == "web_upload":
                clients[platform] = self._init_web_upload_client(platform, config)
        return clients
    
    def publish(
        self,
        audio_path: str,
        script: str,
        metadata: Dict,
        platforms: Optional[List[str]] = None,
        dry_run: bool = False
    ) -> Dict[str, str]:
        """
        一键发布到所有或指定平台
        
        Args:
            audio_path: 音频文件路径
            script: 完整文案
            metadata: {
                "title": "播客标题",
                "topic": "作文题目",
                "grade": "高中",
                "shownotes": "节目简介",
                "tags": ["高考", "作文"],
                "cover_image": "封面图路径",
                "mindmap": "思维导图路径"
            }
            platforms: 指定平台列表（None表示全部）
            dry_run: 试运行模式（不实际发布）
        
        Returns:
            各平台发布结果 {"xiaoyuzhou": "success", "ximalaya": "failed: ..."}
        """
        results = {}
        target_platforms = platforms or list(self.platform_clients.keys())
        
        # 1. 上传音频到云存储（获取公网链接）
        audio_url = self._upload_audio(audio_path) if not dry_run else "mock_url"
        
        for platform_name in target_platforms:
            if platform_name not in self.platform_clients:
                results[platform_name] = f"skipped: platform not configured"
                continue
            
            try:
                client = self.platform_clients[platform_name]
                if dry_run:
                    results[platform_name] = "dry_run_success"
                    continue
                
                # 调用平台客户端发布
                result = client.publish(
                    audio_url=audio_url,
                    script=script,
                    metadata=metadata
                )
                results[platform_name] = f"success: {result.get('url', 'published')}"
            except Exception as e:
                results[platform_name] = f"failed: {str(e)}"
                logger.error(f"发布到 {platform_name} 失败: {e}")
        
        return results
